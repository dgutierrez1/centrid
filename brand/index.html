Implementation Guide

CSS Custom Properties

/* Centrid Design Tokens */ :root { /* Brand Colors */ --color-centrid-primary: #7C3AED; --color-centrid-dark: #1a1a1a; --color-centrid-light: #f8f9fa; /* Agent Colors - Harmonized Purple */ --color-create: #7C3AED; --color-edit: #A855F7; --color-research: #6366F1; } /* Dark Mode */ @media (prefers-color-scheme: dark) { :root { /* Dark Mode Agent Colors */ --color-create: #8B5CF6; --color-edit: #C084FC; --color-research: #818CF8; } }

Complete Color Variables

/* Light Mode Colors */ :root { --color-centrid-primary: #7C3AED; --color-create: #7C3AED; --color-edit: #A855F7; --color-research: #6366F1; } /* Dark Mode Colors */ @media (prefers-color-scheme: dark) { :root { --color-create: #8B5CF6; --color-edit: #C084FC; --color-research: #818CF8; } /* Spacing */ --space-sm: 8px; --space-md: 16px; --space-lg: 24px; }

React Components

// Centrid Button Component const CentridButton = ({ variant = "primary", agent, children, ...props }) => ( <button className={`btn btn--${variant} ${agent ? `btn--agent-${agent}` : ""}`} {...props} > {children} </button> ); // Usage Examples <CentridButton>Primary Action</CentridButton> <CentridButton variant="outline">Outline Button</CentridButton> <CentridButton variant="minimal">Minimal Button</CentridButton> <CentridButton variant="ghost">Ghost Button</CentridButton> <CentridButton variant="outline" agent="create">📝 Ask Create Agent</CentridButton>

Mobile-First CSS

/* Mobile-First Responsive Design */ .centrid-component { /* Mobile styles (320px+) */ font-size: var(--font-body-md); padding: var(--space-sm); } @media (min-width: 768px) { .centrid-component { /* Tablet styles */ font-size: var(--font-body-lg); padding: var(--space-md); } } @media (min-width: 1024px) { .centrid-component { /* Desktop styles */ padding: var(--space-lg); } }
